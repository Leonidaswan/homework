cmake_minimum_required(VERSION 3.25)

project(gotocell_qc VERSION 0.1 LANGUAGES CXX)

find_package(Qt6 REQUIRED COMPONENTS Widgets)

qt_standard_project_setup()

qt_add_executable(gotocell_qc)

target_sources(gotocell_qc
    PRIVATE
        main.cpp
        gotocelldialog.cpp
        gotocelldialog.h
        gotocelldialog.ui

    PUBLIC
        
)

target_compile_features(gotocell_qc PRIVATE cxx_std_20)

target_link_libraries(gotocell_qc
    PRIVATE 
        Qt6::Widgets
)
